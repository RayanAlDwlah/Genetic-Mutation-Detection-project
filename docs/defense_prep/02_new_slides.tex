\begin{frame}{The problem in one slide}
  \begin{block}{Missense variants at clinical scale}
    \begin{itemize}
      \item A point mutation that swaps one amino acid in a protein
      \item Most common coding variant in the human genome
      \item Most are benign; a clinically-important minority cause disease
      \item Distinguishing the two is the central problem of clinical genetics
    \end{itemize}
  \end{block}

  \vspace{0.3em}
  \begin{alertblock}{Why prior work overstates performance}
    \begin{enumerate}
      \item \textbf{Definitional circularity} --- features derived from the label
            (e.g.\ \texttt{is\_common}) inflate metrics
      \item \textbf{Meta-predictor contamination} --- using REVEL/CADD as features
            in a ClinVar benchmark
      \item \textbf{Paralog leakage} --- KRT1 in train + KRT14 in test
            $\Rightarrow$ 80\% sequence identity $\Rightarrow$ model has seen most of the test
    \end{enumerate}
    \vspace{0.4em}
    \footnotesize
    Exome sequencing yields $\sim$20{,}000 variants per patient; human curation
    is $\sim$few hundred per day. \emph{Scale demands automation; automation
    demands honest evaluation.}
  \end{alertblock}
\end{frame}

\begin{frame}{Project objectives}

    \midrule
    SIFT                                    & 2003 & conservation only          & legacy, no guards & 0.620 \\

    \rowcolor{kkugold!20}

    \bottomrule
  \end{tabular}

  \vspace{0.8em}
  \scriptsize
  \textsuperscript{*}\,AlphaMissense is ClinVar-calibrated; on a ClinVar-derived test set its numbers partly overlap its training signal.\\
  \,REVEL omitted: its 13 base learners are individually trained on ClinVar, creating compound contamination if used as a feature.
\end{frame}

\begin{frame}{Why our numbers are different}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{What only we report}
    \begin{itemize}
      \item Paralog-family-disjoint split (CI-enforced)
      \item Three-source leakage audit with before/after numbers
      \item External validation on \textsc{denovo-db}
      \item Bootstrap 95\% CIs on every metric + paired bootstrap on ablations
      \item Calibration deep-dive (ECE = 0.011)
      \item Docker + lock file reproduction to six decimals
    \end{itemize}

    \column{0.5\textwidth}
    \textbf{Honest framing}
    \begin{itemize}
      \item $+0.245$ PR-AUC over SIFT
      \item $+0.137$ PR-AUC over PolyPhen-2
      \item $-0.025$ PR-AUC vs.\ AlphaMissense
      \item AM trains on ClinVar-derived labels; our test shares that distribution, so the AM gap is likely smaller than reported once methodology is matched
    \end{itemize}
  \end{columns}

  \vspace{0.6em}
  \begin{exampleblock}{Take-away}
    Under matched methodology our model beats classical tools cleanly and
    trails AlphaMissense by a margin smaller than the bias from AM's ClinVar
    calibration.
  \end{exampleblock}
\end{frame}

\begin{frame}{Future work}
  \begin{enumerate}\setlength{\itemsep}{0.5em}
    \item \textbf{Larger-checkpoint ESM-2.}
          Upgrade to \texttt{esm2\_t33\_650M\_UR50D} ($19{\times}$ parameters).
          Expected to lift the OOD ceiling; the $\Delta$ROC = $-0.13$ on
          unseen families is the binding constraint of the current work.

    \item \textbf{AlphaFold2 structural features.}
          pLDDT, SASA, DSSP secondary structure, residue depth ---
          orthogonal to PLM signal, complementary to conservation.

    \item \textbf{Deep mutational scanning external validation.}
          ProteinGym fitness measurements as a third external dataset,
          independent of both ClinVar and \textsc{denovo-db}.

    \item \textbf{Hybrid stacking + ACMG integration.}
          Meta-learner combining XGBoost + ESM-2 + AlphaMissense, packaged
          as an ACMG-compatible plug-in for clinical workflows.
  \end{enumerate}

  \vspace{0.6em}
  \begin{block}{Already completed during this project}
    Training-time ESM-2 LLR integration:
    test PR-AUC $0.827 \to 0.865$ ($\Delta = +0.031,\ p < 10^{-4}$);
    OOD family-holdout ceiling at 35M characterised.
  \end{block}
\end{frame}

\begin{frame}{This talk in one minute}
  \begin{block}{The problem}
    Published variant predictors report ROC-AUC above 0.92.
    Most do not survive external validation.
  \end{block}

  \begin{block}{What we did}
    Built a classifier for 195k missense variants.
    Found three contamination sources in our own pipeline.
    Removed them. Watched PR-AUC fall from 0.955 to 0.838.
    Re-scored published baselines under the same rules.
    Validated externally on \textsc{denovo-db}.
  \end{block}

  \begin{exampleblock}{What we found}
    Audited numbers are smaller than inflated ones --- and they replicate.
    The audit itself is the contribution.
  \end{exampleblock}
