\section{Introduction}

% ───────────── SLIDE 2 — Problem statement ─────────────
\begin{frame}{Problem statement: missense variants at clinical scale}
  \footnotesize
  \begin{block}{Missense variants at clinical scale}
    \begin{itemize}\setlength{\itemsep}{0.05em}
      \item A point mutation that swaps one amino acid in a protein.
      \item Most common coding variant in the human genome.
      \item Most are benign; a clinically-important minority cause disease.
      \item Distinguishing the two is the central problem of clinical
            genetics.
    \end{itemize}
  \end{block}

  \vspace{0.2em}
  \begin{alertblock}{Why prior work overstates performance}
    \begin{enumerate}\setlength{\itemsep}{0.05em}
      \item \textbf{Definitional circularity} --- features derived from
            the label (e.g.\ \texttt{is\_common}) inflate metrics.
      \item \textbf{Meta-predictor contamination} --- using REVEL/CADD
            as features in a ClinVar benchmark.
      \item \textbf{Paralog leakage} --- KRT1 in train + KRT14 in test
            $\Rightarrow$ 80\% sequence identity $\Rightarrow$ model
            has seen most of the test.
    \end{enumerate}
    \vspace{0.2em}
    \scriptsize
    Exome: $\sim$20{,}000 variants/patient; human curation $\sim$a few
    hundred/day. \emph{Scale demands automation; automation demands
    honest evaluation.}
  \end{alertblock}
\end{frame}

% ───────────── SLIDE 3 — Research objectives + scope ─────────────
\begin{frame}{Research objectives + scope}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Data + methodology}
    \begin{enumerate}
      \item Assemble a paralog-disjoint corpus of 195k missense
            variants.
      \item Eliminate three specific leakage sources.
      \item Train and calibrate a defensible baseline.
      \item Re-score published tools on our own test split.
    \end{enumerate}

    \column{0.5\textwidth}
    \textbf{Validation + delivery}
    \begin{enumerate}\setcounter{enumi}{4}
      \item Evaluate externally on \textsc{denovo-db}.
      \item Prove orthogonality of zero-shot ESM-2 signal.
      \item Publish everything: code, tests, Docker, 10 narrative
            notebooks, Streamlit demo, academic report.
    \end{enumerate}
  \end{columns}

  \vspace{0.7em}
  \centering
  \footnotesize
  \emph{Every objective maps to at least one CI-enforced check.}
\end{frame}

\section{Related Work}

% ───────────── SLIDE 4 — Classical tools ─────────────
\begin{frame}{Classical tools (2003--2010): conservation-based predictors}
  \footnotesize
  \begin{tabular}{@{} p{1.7cm} p{2.7cm} p{2.9cm} p{2.9cm} @{}}
    \toprule
    \textbf{Tool (year)} & \textbf{Algorithm} & \textbf{\textcolor{green!40!black}{Pros}} & \textbf{\textcolor{red!70!black}{Cons}} \\
    \midrule
    \textbf{SIFT} (2003) &
    Conservation only --- PSSM from PSI-BLAST &
    Instant via VEP REST; baseline of $>$95\% clinical tools &
    No AA chemistry; arbitrary 0.05 threshold; weak on slow-evolving sites \\
    \midrule
    \textbf{PolyPhen-2} (2010) &
    Naïve Bayes on conservation + chemistry + 3D struct.\ (HumDiv) &
    Biochemical context (Grantham, BLOSUM); standard in clinical reports &
    \alert{HumDiv overlaps ClinVar} $\Rightarrow$ contamination on ClinVar benchmarks \\
    \bottomrule
  \end{tabular}

  \vspace{0.6em}
  \begin{exampleblock}{Where we re-score them}
    On our paralog-disjoint test (own-coverage):
    SIFT $= 0.620$ PR-AUC (96\,\% cov.), PolyPhen-2 $= 0.728$ (93\,\% cov.). See Section~9.
  \end{exampleblock}
\end{frame}

% ───────────── SLIDE 5 — Gradient-boosted ensembles ─────────────
\begin{frame}{Gradient-boosted ensembles (2016--2024): supervised meta-predictors}
  \scriptsize
  \begin{tabular}{@{} p{1.5cm} p{2.6cm} p{3cm} p{3cm} @{}}
    \toprule
    \textbf{Tool (year)} & \textbf{Algorithm} & \textbf{\textcolor{green!40!black}{Pros}} & \textbf{\textcolor{red!70!black}{Cons}} \\
    \midrule
    \textbf{REVEL} (2016) &
    Random forest ensemble of 13 prediction tools &
    Very high reported AUC ($\sim$0.95); widely adopted &
    \alert{Inputs ClinVar-trained} $\Rightarrow$ circular when benchmarked on ClinVar \\
    \midrule
    \textbf{CADD} (2019) &
    Logistic/RF on simulated negatives + observed positives &
    Single model for all variant types (incl.\ non-coding) &
    Scores uncalibrated; training labels are simulated \\
    \midrule
    \textbf{VARITY} (2021) &
    XGBoost with noise-weighted training &
    Handles label noise explicitly &
    \alert{Gene-level split only} (no paralog protection) \\
    \midrule
    \textbf{MAGPIE} (2024) &
    Gradient boosting on multi-omics features &
    Recent, multi-modal feature space &
    Paralog-split status undocumented in the publication \\
    \bottomrule
  \end{tabular}

  \vspace{0.4em}
  \begin{exampleblock}{Why we ban these as features}
    \texttt{verify\_no\_leakage.py} excludes REVEL, CADD, ClinPred, M-CAP, MetaSVM ---
    using meta-predictors trained on ClinVar would inflate PR-AUC by $\sim$0.07.
  \end{exampleblock}
\end{frame}

% ───────────── SLIDE 6 — Protein language models ─────────────
\begin{frame}{Protein language models (2021--2023): unsupervised representations}
  \footnotesize
  \begin{tabular}{@{} p{1.7cm} p{2.7cm} p{2.9cm} p{2.9cm} @{}}
    \toprule
    \textbf{Tool (year)} & \textbf{Algorithm} & \textbf{\textcolor{green!40!black}{Pros}} & \textbf{\textcolor{red!70!black}{Cons}} \\
    \midrule
    \textbf{EVE} (2021) &
    Variational auto-encoder, one per gene ($\sim$3{,}000 genes) &
    Gene-specific deep model; captures epistasis &
    Limited gene coverage; slow per-gene training \\
    \midrule
    \textbf{ESM-1b / ESM-2} (2023, zero-shot) &
    Masked-AA log-likelihood ratio from 250M-sequence corpus &
    \textbf{Fully unsupervised} $\Rightarrow$ uncontaminated by ClinVar &
    Lower AUC at small scale (we use \texttt{esm2\_t12\_35M}) \\
    \midrule
    \textbf{AlphaMissense} (2023) &
    Structural PLM calibrated against ClinVar at release &
    SOTA reported AUC ($\sim$0.96 on own benchmark) &
    \alert{Calibrated on ClinVar} $\Rightarrow$ comparison contamination on ClinVar tests \\
    \bottomrule
  \end{tabular}

  \vspace{0.4em}
  \begin{exampleblock}{The PLM choice in our pipeline}
    ESM-2 is the only one provably unsupervised $\Rightarrow$ we integrate its LLR
    as a Phase-2.1 feature (Section~8). Lift on internal test:
    $\Delta$PR-AUC $=+0.0313$, $p<10^{-4}$.
  \end{exampleblock}
\end{frame}

% ───────────── SLIDE 9 — Part 1: Preprocessing (merged) ─────────────
\begin{frame}{Part 1 --- Data preprocessing}
  \footnotesize
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{\textcolor{kkuGreen}{Sources}} (joined on GRCh38 \texttt{chr:pos:ref:alt})
    \begin{itemize}\setlength{\itemsep}{0.1em}
      \item \textsc{ClinVar} --- labels (review star $\geq 2$)
      \item \textsc{gnomAD} v2.1 + v4 --- AF, AC, AN
      \item \textsc{dbNSFP} 5.3 --- conservation, AA chemistry, Pfam
      \item \textsc{gnomAD} v2.1.1 --- gene-level constraint (pLI, LOEUF, mis\_z)
    \end{itemize}

    \vspace{0.4em}
    \textbf{\textcolor{kkuGreen}{Output}} 195{,}098 missense variants $\rightarrow$
    Part~2 as \texttt{splits/\{train,val,test\}.parquet}.

    \column{0.52\textwidth}
    \textbf{\textcolor{kkuGreen}{Two critical filters}}

    \begin{block}{Missense-only}
      Drop rows with null \texttt{ref\_aa} OR null \texttt{alt\_aa}
      $\Rightarrow$ remove \textbf{$\sim$88K non-missense rows}.
      Pre-audit: 64\% of pathogenic rows had null AA $\Rightarrow$
      model learned ``null $\Rightarrow$ pathogenic''.
      \alert{Fix alone drops PR-AUC $0.955 \rightarrow 0.819$.}
    \end{block}

    \begin{block}{Paralog-aware split}
      Family-level via \texttt{assign\_gene\_family}
      (16 regex + trailing-digit fallback).
      \textbf{15{,}479 genes $\rightarrow$ 7{,}851 families},
      zero shared across train/val/test (CI-enforced).
    \end{block}
  \end{columns}
\end{frame}

% ───────────── SLIDE 10 — Part 2: Training + leakage + calibration ─────────────
\begin{frame}{Part 2 --- Training, leakage audit, calibration}
  \scriptsize
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{\textcolor{kkuGreen}{Leakage gate (6 CI-enforced invariants)}}
    \begin{enumerate}\setlength{\itemsep}{0.04em}
      \item No banned features (\texttt{is\_common, chr, ref, alt})
      \item 100\,\% missense-only rows in training
      \item Zero shared gene families (gene + family disjoint)
      \item Pathogenic-rate gap $< 0.08$ across splits
      \item Phase-2.1 ESM-2 features pass banned-upstream-meta check
      \item ESM-2 split integrity (no test/val overlap in score parquet)
    \end{enumerate}
    \emph{Every push to \texttt{main} runs all six;
    failure blocks the merge.}

    \vspace{0.6em}
    \textbf{\textcolor{kkuGreen}{Five-stage leakage journey}} (raw PR-AUC)
    $0.955 \rightarrow 0.819 \rightarrow 0.816 \rightarrow 0.835 \rightarrow 0.836$.
    Calibrated headline: $0.827$.

    \column{0.5\textwidth}
    \textbf{\textcolor{kkuGreen}{Model}}
    XGBoost + Optuna TPE (multivariate grouping), 40 trials,
    PR-AUC objective, median pruner after 200 rounds.

    \vspace{0.4em}
    \textbf{\textcolor{kkuGreen}{Calibration}} Isotonic on validation;
    Murphy: $\text{Brier} = \text{Rel} - \text{Res} + \text{Unc}$.
    \begin{tabular}{@{} l r r r @{}}
      \toprule
      & Brier & Rel.\ & ECE \\
      \midrule
      Raw      & 0.088 & 0.0054 & 0.054 \\
      Platt    & 0.083 & 0.0011 & 0.029 \\
      \textbf{Isotonic} & \textbf{0.083} & \textbf{0.0002} & \textbf{0.011} \\
      \bottomrule
    \end{tabular}
    \emph{Reliability drops 23$\times$; resolution unchanged
    $\Rightarrow$ discrimination preserved.}
  \end{columns}
\end{frame}

% ───────────── SLIDE 11 — Part 3: Evaluation framework ─────────────
\begin{frame}{Part 3 --- Evaluation framework}
  \footnotesize
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{\textcolor{kkuGreen}{Internal evaluation}}
    \begin{itemize}\setlength{\itemsep}{0.12em}
      \item Paralog-disjoint test split — \textbf{28{,}098 variants}.
      \item 95\,\% bootstrap CI on every metric (1{,}000 reps, seed 42):
            ROC-AUC, PR-AUC, F1, MCC, Brier.
      \item Murphy decomposition for calibration
            (Reliability, Resolution, Uncertainty).
      \item \textbf{Paired bootstrap on $\Delta$} for ablations
            (n = 28{,}098, same resampled indices for both models).
    \end{itemize}

    \column{0.5\textwidth}
    \textbf{\textcolor{kkuGreen}{External evaluation}}
    \begin{itemize}\setlength{\itemsep}{0.12em}
      \item \textsc{denovo-db} family-holdout — \textbf{n = 201}
            unseen gene families.
      \item Pre- vs.\ post-constraint paired bootstrap
            (recovered checkpoint at \texttt{f8ab464}).
      \item Permutation test for $\text{AUC} > 0.5$
            (10{,}000 reps, seed 42).
      \item Calibration transfer audit: does isotonic from
            ClinVar val transfer to denovo-db?
    \end{itemize}
  \end{columns}

  \vspace{0.5em}
  \begin{exampleblock}{Why this framework}
    Internal CIs guard against single-split optimism;
    paired tests guard against unfair $\Delta$ comparisons;
    external denovo-db guards against ClinVar-shaped overfit.
  \end{exampleblock}
\end{frame}

\begin{frame}{Backup --- denovo-db: paired bootstrap vs permutation test}

  \begin{block}{Two separate tests --- different null hypotheses}

    \begin{itemize}\setlength{\itemsep}{0.1em}

      \item \textbf{Paired bootstrap} ($n_{\text{boot}}=1{,}000$, seed 42,

        Tests \emph{``did adding constraint features help, on the same 201 variants?''}
    \end{itemize}

    \textbf{Slice} & \textbf{Pre-constraint} & \textbf{Post-constraint} & \textbf{Paired $\Delta$} \\
